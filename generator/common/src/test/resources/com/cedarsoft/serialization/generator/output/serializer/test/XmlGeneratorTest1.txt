package com.cedarsoft.serialization.generator.parsing.test;

import com.cedarsoft.Version;
import com.cedarsoft.serialization.AbstractXmlVersionTest2;
import com.cedarsoft.serialization.Serializer;
import com.cedarsoft.serialization.VersionEntry;
import org.junit.Assert;
import org.junit.experimental.theories.DataPoint;

public class HouseSerializerVersionTest
    extends AbstractXmlVersionTest2 <House>
{

    @DataPoint
    public final static VersionEntry ENTRY1 = HouseSerializerVersionTest.create(Version.valueOf(1, 0, 0), HouseSerializerVersionTest.class.getResourceAsStream("HouseSerializerVersionTest.1.xml"));

    @Override
    protected Serializer<House> getSerializer()
        throws Exception
    {
        return new HouseSerializer();
    }

    @Override
    protected void verifyDeserialized(House deserialized, Version version)
        throws Exception
    {
        Assert.assertEquals(deserialized.getRooms(), "daValue");
        Assert.assertEquals(deserialized.getFrontDoor(), "daValue");
    }

}

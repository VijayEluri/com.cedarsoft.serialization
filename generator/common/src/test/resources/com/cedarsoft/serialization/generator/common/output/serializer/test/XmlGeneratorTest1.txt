package com.cedarsoft.serialization.generator.common.parsing.test;

import com.cedarsoft.serialization.Serializer;
import com.cedarsoft.serialization.test.utils.AbstractXmlVersionTest2;
import com.cedarsoft.serialization.test.utils.VersionEntry;
import com.cedarsoft.version.Version;
import org.junit.Assert;
import org.junit.experimental.theories.DataPoint;

public class HouseSerializerVersionTest
    extends AbstractXmlVersionTest2 <House>
{

    @DataPoint
    public final static VersionEntry ENTRY1 = HouseSerializerVersionTest.create(Version.valueOf(1, 0, 0), HouseSerializerVersionTest.class.getResource("House_1.0.0_1.xml"));

    @Override
    protected Serializer<House> getSerializer()
        throws Exception
    {
        return new HouseSerializer();
    }

    @Override
    protected void verifyDeserialized(House deserialized, Version version)
        throws Exception
    {
        Assert.assertEquals("daValue", deserialized.getRooms());
        Assert.assertEquals("daValue", deserialized.getFrontDoor());
    }

}
